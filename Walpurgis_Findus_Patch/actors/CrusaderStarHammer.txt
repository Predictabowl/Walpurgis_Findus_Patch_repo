	//Aria is a recurring character of mine, a tomboyish, short, silver haired gal.
	//She actually normally uses a Spear, but sometimes uses a large Hammer too,
	//due her extraordinary super-strength. She was created for a long running D&D
	//campaign I GM'ed in 2009, as an NPC Paladin of the Star Traveller God, Al Kinah. 
	//
	//"Al Kinah" is lit. "The Shining One" in the language of the Hal-Castemourites.
	//Their culture is a mix of Mayan and Arabian influences, and funnily enough
	//a lot of the inspiration for Parias' god and powers comes from Al Kinah!

			}

				}

		A_TakeInventory("Mana1W", 2,TIF_NOTAKEINFINITE);
		A_TakeInventory("Mana2W", 2,TIF_NOTAKEINFINITE);

			}
			else
			{
			A_SpawnItemEx("CometShotA",30,0,40,36,0,0,0,SXF_SETMASTER);
			}}

		ST8H M 3
		ST8H N 3 A_Refire("AltfireBack")

	AltFireBack:
		ST8H O 3
		Goto AltFire
	}

	}
